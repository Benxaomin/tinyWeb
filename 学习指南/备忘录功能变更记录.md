## 功能概述

在页面右侧栏的「时间胶囊」卡片内，新增一个**备忘录**功能：
- 时间胶囊右上角新增一个 📋 图标按钮，带红色角标显示未完成事项数量
- 点击图标，面板从**时间胶囊右侧**弹出（不占用时间胶囊内部空间）
- 支持「生活」「学习」「*** 重要」三个分类标签页切换
- 「*** 重要」用于记录长期重要任务
- 支持添加、完成勾选、双击编辑、删除事项
- 数据自动保存到浏览器 `localStorage`，刷新页面不丢失
- 每天自动归档当天任务到「归档箱」，可翻看历史日记

**典型 CRUD 操作示例**：

```javascript
// ✅ CREATE - 添加事项
function addTodo(text, category) {
    todos[category].push({ text: text, done: false });  // 往数组末尾加一项
    saveTodos();          // 存入 localStorage
    renderTodos();        // 重新渲染列表
    updateBadge();        // 更新角标数字
}

// ✅ READ - 渲染列表
function renderTodos() {
    var list = document.getElementById('todoList');  // 找到 <ul> 容器
    list.innerHTML = '';                            // 清空现有内容

    todos[currentTab].forEach(function(item, index) {  // 遍历当前分类的所有事项
        var li = document.createElement('li');         // 创建 <li>
        li.className = item.done ? 'todo-item done' : 'todo-item';  // 已完成加 done 类
        li.innerHTML = '<span class="todo-check"></span>' +
                       '<span class="todo-text">' + item.text + '</span>' +
                       '<button class="todo-del">✕</button>';
        list.appendChild(li);                         // 插入到 <ul>
    });
}

// ✅ UPDATE - 切换完成状态
function toggleTodo(index, category) {
    todos[category][index].done = !todos[category][index].done;  // 取反：true↔false
    saveTodos();
    renderTodos();
    updateBadge();
}

// ✅ DELETE - 删除事项
function deleteTodo(index, category) {
    todos[category].splice(index, 1);  // 从 index 位置删除 1 个元素
    saveTodos();
    renderTodos();
    updateBadge();
}
```

### 🧩 进阶概念详解

#### 1️⃣ 什么是 IIFE？为什么要用它？

```javascript
// 这就是 IIFE（Immediately Invoked Function Expression）
(function() {
    'use strict';

    // 所有变量都声明在这里面 —— 不会污染全局作用域
    var todos = {};
    var currentTab = 'life';

    function loadTodos() { ... }
    function saveTodos() { ... }

    // 最后调用初始化
    initTodo();

})();  // ← 定义完立刻执行
```

**好处**：`todos`、`currentTab` 这些变量不会和其他代码冲突。如果不用 IIFE，它们会变成全局变量，可能覆盖别人的同名变量。

#### 2️⃣ 什么是事件委托（Event Delegation）？

```javascript
// ❌ 不好的做法：每个按钮都绑定事件（内存浪费）
items.forEach(function(item, i) {
    item.querySelector('.todo-del').addEventListener('click', function() {
        deleteTodo(i);
    });
});

// ✅ 好的做法：只在父元素上绑定一次事件
document.getElementById('todoList').addEventListener('click', function(e) {
    if (e.target.classList.contains('todo-del')) {   // 点击的是删除按钮？
        var index = Array.from(e.target.parentNode.parentNode.children)
                        .indexOf(e.target.parentNode);
        deleteTodo(index);
    }
    if (e.target.classList.contains('todo-check')) { // 点击的是勾选框？
        var index = Array.from(e.target.parentNode.parentNode.children)
                        .indexOf(e.target.parentNode);
        toggleTodo(index);
    }
});
```

**原理**：点击子元素时，事件会"冒泡"到父元素。在父元素统一处理，不管有多少个子元素都只需要一个监听器。

#### 3️⃣ localStorage 的限制你知道吗？

| 特性 | 说明 |
|------|------|
| 容量限制 | 约 5MB（不同浏览器略有差异） |
| 只能存字符串 | 对象/数组需要 `JSON.stringify()` / `JSON.parse()` |
| 同源策略 | 只有同域名/端口的页面才能访问 |
| 无过期时间 | 数据永久保存，除非手动清除或用户清理浏览器 |
| 同步操作 | 读写时会阻塞主线程（数据量小时无感） |

#### 4️⃣ 归档逻辑的日期计算技巧

```javascript
// 获取今天的日期字符串："2026-04-02"
function getTodayStr() {
    var d = new Date();
    return d.getFullYear() + '-' +
           String(d.getMonth() + 1).padStart(2, '0') + '-' +  // 月、日补零
           String(d.getDate()).padStart(2, '0');
}

// 获取前一天日期
function prevDay(dateStr) {
    var d = new Date(dateStr);
    d.setDate(d.getDate() - 1);  // JavaScript Date 自动处理跨月/跨年
    return getTodayStr.call({ getDate: () => d.getDate(), ... }); // 简化写法
    // 或者更直观的方式：
    // return d.getFullYear() + '-' + ...
}
```
